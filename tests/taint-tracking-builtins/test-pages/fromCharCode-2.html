  if (typeof J$$ !== 'undefined' && J$$.wrapTaint) {

    // Set the src of the new script element
    scriptEle.src = taintedSrc;

    document.body.appendChild(scriptEle);

    console.error("J$$ is not defined or does not have wrapTaint method.");
  }
